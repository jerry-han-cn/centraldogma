17  package com.linecorp.centraldogma.server;
18  
19  import static com.google.common.base.MoreObjects.firstNonNull;
20  import static org.apache.curator.shaded.com.google.common.base.Preconditions.checkArgument;
21  
22  import javax.annotation.Nullable;
23  
24  import com.cronutils.model.Cron;
25  import com.cronutils.model.CronType;
26  import com.cronutils.model.definition.CronDefinitionBuilder;
27  import com.cronutils.parser.CronParser;
28  import com.fasterxml.jackson.annotation.JsonCreator;
29  import com.fasterxml.jackson.annotation.JsonProperty;
30  
31  import com.linecorp.centraldogma.server.storage.repository.Repository;
32  
33  /**
34   * A configuration for {@link Repository} garbage collection.
35   */
36  public final class RepositoryGarbageCollectionConfig {
37  
38      private static final String DEFAULT_SCHEDULE = "0 0 * * * ?"; // Every day
39      private static final CronParser cronParser =
40              new CronParser(CronDefinitionBuilder.instanceDefinitionFor(CronType.QUARTZ));
41  
42      private final int minNumNewCommits;
43      private final Cron schedule;
44  
45      /**
46       * Creates a new instance.
47       */
48      @JsonCreator
49      public RepositoryGarbageCollectionConfig(@JsonProperty("minNumNewCommits") int minNumNewCommits,
50                                               @JsonProperty("schedule") @Nullable String schedule) {
51          checkArgument(minNumNewCommits > 0, "minNumNewCommits: %s (expected: > 0)", minNumNewCommits);
52  
53          this.minNumNewCommits = minNumNewCommits;
54          this.schedule = cronParser.parse(firstNonNull(schedule, DEFAULT_SCHEDULE));
55      }
56  
57      /**
58       * Returns the minimum required number of commits newly added to run a garbage collection.
59       */
60      public int minNumNewCommits() {
61          return minNumNewCommits;
62      }
63  
64      /**
65       * Returns the schedule when garbage collections is supposed to be triggered.
66       */
67      public Cron schedule() {
68          return schedule;
69      }
70  }
